class Solution {
private:
    int mem[30+5];
    int dp(int n){
        if(n == 1 || n == 2) return 1;
        if(mem[n]) return mem[n];
        return mem[n] = dp(n-1) + dp(n-2);
    }
public:
    int fib(int n) {
        if(n == 0) return 0;
        memset(mem, 0, sizeof(mem));
        return dp(n);
    }
};
